\frametitle{Cos'è l'HTML?}
  \textbf{HTML}\only<2>{ (HyperText Markup Language)}\only<3->{ è un linguaggio di markup per ipertesti.}
  \only<4-5>{
    \begin{description}\centering
      \item[Linguaggio di markup] Descrive il documento.
      \item<5->[Ipertesto] Testo con collegamenti ad altri ipertesti.
    \end{description}
  }
  \only<6->{\par Creato a inizi anni '90 al CERN come strumento di condivisione paper fra i vari istituti di ricerca.}
}
